route: /api/token/map/favorite
method: GET
parameters:
    query:
        -   name: offset
            type: integer
            description: The offset to start from
            required: false
            default: 0
        -   name: limit
            type: integer
            description: The maximum number of tracks to return
            required: false
            minimum: 1
            maximum: 100
        -   name: sort
            type: string
            description: The sort order of the tracks
            required: false
            default: "date"
            enum:
                - "date"
                - "name"
        -   name: order
            type: string
            description: The order of the tracks
            required: false
            default: "desc"
            enum:
                - "asc"
                - "desc"
        -   name: mapType
            type: string
            description: The map type filter
            required: false
            enum:
                - "TrackMania\\TM_Race"
        -   name: playable
            type: integer
            description: Whether the map is validated and playable
            required: false
            enum:
                - 0
                - 1
---
